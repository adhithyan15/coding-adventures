2 @moduledoc """
3 BLAKE2b cryptographic hash function (RFC 7693), from scratch.
4
5 BLAKE2b is a modern hash that is:
6
7 * Faster than MD5 on 64-bit hardware.
8 * As secure as SHA-3 against known attacks.
9 * Variable output length (1..64 bytes).
10 * Keyed in a single pass (replaces HMAC-SHA-512).
11 * Parameterized with salt and personalization (16 bytes each).
12
13 It underlies libsodium, WireGuard, Noise Protocol, IPFS content addressing,
14 and -- within this repo -- Argon2.
15
16 Elixir integers are arbitrary precision, so every 64-bit add and XOR masks
17 with `0xFFFFFFFFFFFFFFFF` to stay inside a single machine word. This is
18 analogous to the Ruby and Python ports; Go, Rust, and Swift use native
19 wrapping `u64` arithmetic instead.
20
21 Sequential mode only. Tree hashing, BLAKE2s, BLAKE2bp, BLAKE2sp,
22 BLAKE2Xb, and BLAKE3 are out of scope.
23
24 > Educational implementation. Use a vetted library for real crypto.
25 """

76 Create a new streaming hasher.
77
78 Options:
79
80 * `:digest_size` -- integer in `1..64` (default `64`)
81 * `:key` -- binary of length `0..64` (default `<<>>`)
82 * `:salt` -- binary of length `0` or `16` (default `<<>>`)

85 @spec new(keyword()) :: %Hasher{}
86 def new(opts \\ []) do
87 digest_size = Keyword.get(opts, :digest_size, 64)
88 key = Keyword.get(opts, :key, <<>>)
89 salt = Keyword.get(opts, :salt, <<>>)
90 personal = Keyword.get(opts, :personal, <<>>)
91 validate!(digest_size, key, salt, personal)
92
93 state = initial_state(digest_size, byte_size(key), salt, personal)
94
95 buffer =

97 <<>>
98 else
99 key <> :binary.copy(<<0>>, @block_size - byte_size(key))
100 end

105 @doc "Feed more bytes into the stream."
106 @spec update(%Hasher{}, binary()) :: %Hasher{}
107 def update(%Hasher{} = h, data) when is_binary(data) do
108 drain(%{h | buffer: h.buffer <> data})
109 end

111 @doc """
112 Finalize and return the digest. Non-destructive: the hasher keeps its
113 state so `digest/1` can be called again or followed by more `update/2`
114 calls.
115 """
116 @spec digest(%Hasher{}) :: binary()
117 def digest(%Hasher{state: state, buffer: buf, byte_count: bc, digest_size: ds}) do
118 padded = buf <> :binary.copy(<<0>>, @block_size - byte_size(buf))
119 total = bc + byte_size(buf)
120 new_state = compress(state, padded, total, true)
121 <<raw::binary-size(64)>> = state_to_bytes(new_state)
122 :binary.part(raw, 0, ds)
123 end

126 @spec hex_digest(%Hasher{}) :: String.t()

169 # Flush every full block except the latest -- at least one byte must remain
170 # in the buffer so the final compression is the one flagged final.
171 defp drain(%Hasher{buffer: buf} = h) when byte_size(buf) > @block_size do
172 <<block::binary-size(@block_size), rest::binary>> = buf
173 bc = h.byte_count + @block_size

180 # Build the parameter-block-XOR-ed starting state. Sequential mode only

182 defp initial_state(digest_size, key_len, salt, personal) do
183 salt_padded = salt <> :binary.copy(<<0>>, 16 - byte_size(salt))
184 personal_padded = personal <> :binary.copy(<<0>>, 16 - byte_size(personal))
185 # Parameter block: ds | kl | fanout=1 | depth=1 | leaf(4)=0 | node_offset(8)=0
186 # | node_depth=0 | inner_length=0 | reserved(14)=0
187 # | salt(16) | personal(16)
188 p =
189 <<digest_size::8, key_len::8, 1::8, 1::8, 0::32, 0::64, 0::8, 0::8, 0::unit(8)-size(14),
190 salt_padded::binary, personal_padded::binary>>

206 # Compression function F. `t` is the 128-bit total byte count so far.
207 # `is_final?` triggers the v[14] inversion for the final block.
208 defp compress(state_tuple, block, t, is_final?) do
209 m = parse_block(block)
210
211 # Build the 16-word working vector v. state[0..7] || IV[0..7].
212 v = state_tuple |> Tuple.to_list() |> Kernel.++(Tuple.to_list(@iv)) |> List.to_tuple()
213
214 v =
215 v
216 |> tuple_update(12, fn x -> bxor(x, t &&& @mask64) end)
217 |> tuple_update(13, fn x -> bxor(x, t >>> 64 &&& @mask64) end)
218
219 v = if is_final?, do: tuple_update(v, 14, fn x -> bxor(x, @mask64) end), else: v
220
221 v = Enum.reduce(0..11, v, fn i, acc -> round_fn(acc, m, elem(@sigma, rem(i, 10))) end)
222
223 # Davies-Meyer feed-forward: state[i] ^= v[i] ^ v[i+8].
224 Enum.reduce(0..7, state_tuple, fn i, acc ->
225 tuple_update(acc, i, fn h -> h |> bxor(elem(v, i)) |> bxor(elem(v, i + 8)) end)
226 end)
227 end

229 # One of twelve rounds: four column G's then four diagonal G's.
230 defp round_fn(v, m, s) do
231 v
232 |> g(0, 4, 8, 12, elem(m, elem(s, 0)), elem(m, elem(s, 1)))
233 |> g(1, 5, 9, 13, elem(m, elem(s, 2)), elem(m, elem(s, 3)))
234 |> g(2, 6, 10, 14, elem(m, elem(s, 4)), elem(m, elem(s, 5)))
235 |> g(3, 7, 11, 15, elem(m, elem(s, 6)), elem(m, elem(s, 7)))
236 |> g(0, 5, 10, 15, elem(m, elem(s, 8)), elem(m, elem(s, 9)))
237 |> g(1, 6, 11, 12, elem(m, elem(s, 10)), elem(m, elem(s, 11)))
238 |> g(2, 7, 8, 13, elem(m, elem(s, 12)), elem(m, elem(s, 13)))
239 |> g(3, 4, 9, 14, elem(m, elem(s, 14)), elem(m, elem(s, 15)))
240 end

242 # BLAKE2b quarter-round G. Mutates v[a], v[b], v[c], v[d] with message
243 # words x and y. Rotation constants (R1..R4) = (32, 24, 16, 63).
244 defp g(v, a, b, c, d, x, y) do
245 va = elem(v, a)
246 vb = elem(v, b)
247 vc = elem(v, c)
248 vd = elem(v, d)
249
250 va = va + vb + x &&& @mask64
251 vd = rotr64(bxor(vd, va), 32)
252 vc = vc + vd &&& @mask64
253 vb = rotr64(bxor(vb, vc), 24)
254 va = va + vb + y &&& @mask64
255 vd = rotr64(bxor(vd, va), 16)
256 vc = vc + vd &&& @mask64
257 vb = rotr64(bxor(vb, vc), 63)
258
259 v
260 |> put_elem(a, va)
261 |> put_elem(b, vb)
262 |> put_elem(c, vc)
263 |> put_elem(d, vd)
264 end

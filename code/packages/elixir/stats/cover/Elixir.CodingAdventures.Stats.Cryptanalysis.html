65 @doc """
66 # Median
67
68 The median is the "middle" value when data is sorted. Unlike the mean,
69 it is robust against outliers. A billionaire walking into a room doesn't
70 change the median income much, but it skews the mean enormously.
71
72 ## Algorithm
73
74 1. Sort the values in ascending order.
75 2. If the count is odd, return the middle element.
76 3. If the count is even, return the average of the two middle elements.
77
78 ## Examples
79
80 iex> CodingAdventures.Stats.Descriptive.median([1, 3, 5])
81 3
82
83 iex> CodingAdventures.Stats.Descriptive.median([2, 4, 4, 4, 5, 5, 7, 9])
84 4.5
85 """
86 def median([]), do: raise(ArgumentError, "Cannot compute median of an empty list")
87
88 def median(values) when is_list(values) do
89 sorted = Enum.sort(values)
90 n = length(sorted)
91 mid = div(n, 2)
92
93 if rem(n, 2) != 0 do
94 # Odd length: return middle element.
95 Enum.at(sorted, mid)
96 else
97 # Even length: average the two middle elements.
98 (Enum.at(sorted, mid - 1) + Enum.at(sorted, mid)) / 2
99 end

120 # Build a frequency map. We iterate the original list to find the
121 # first value with the maximum frequency (preserving insertion order).
122 freq_map =
123 Enum.reduce(values, %{}, fn val, acc ->

133 @doc """
134 # Variance
135
136 Variance measures how spread out the data is from the mean.
137
138 ## Formula
139
140 variance = sum((x_i - mean)^2) / d
141
142 Where d is:
143 - n if `population: true` (the entire population)
144 - n-1 if `population: false` (the default — Bessel's correction for samples)
145
146 ## Why n-1? (Bessel's Correction)
147
148 When computing variance from a sample, the sample mean is "pulled toward"
149 the data points, systematically underestimating the true spread. Dividing
150 by n-1 compensates for this bias.
151
152 ## Examples
153
154 iex> CodingAdventures.Stats.Descriptive.variance([2, 4, 4, 4, 5, 5, 7, 9])
155 4.571428571428571
156
157 iex> CodingAdventures.Stats.Descriptive.variance([2, 4, 4, 4, 5, 5, 7, 9], population: true)
158 4.0
159 """
160 def variance(values, opts \\ [])
161
162 def variance([], _opts), do: raise(ArgumentError, "Cannot compute variance of an empty list")
163
164 def variance(values, opts) when is_list(values) do
165 population = Keyword.get(opts, :population, false)
166 n = length(values)
167
168 if not population and n < 2 do
169 raise ArgumentError, "Sample variance requires at least 2 values"
170 end
171
172 avg = mean(values)
173
174 sum_sq_dev =
175 Enum.reduce(values, 0.0, fn val, acc ->
176 acc + :math.pow(val - avg, 2)
177 end)
178
179 divisor = if population, do: n, else: n - 1
180 sum_sq_dev / divisor
181 end
182
183 @doc """
184 # Standard Deviation
185
186 The standard deviation is the square root of the variance. While variance
187 is in "squared units," standard deviation brings us back to the original
188 units, making it more interpretable.
189
190 ## The 68-95-99.7 Rule
191
192 For normally distributed data:
193 - ~68% of values fall within 1 standard deviation of the mean
194 - ~95% fall within 2 standard deviations
195 - ~99.7% fall within 3 standard deviations
196 """
197 def standard_deviation(values, opts \\ []) do
198 :math.sqrt(variance(values, opts))
199 end

249
250 Counts how many times each letter (A-Z) appears in the text.

252

259 text
260 |> String.upcase()
261 |> String.graphemes()
262 |> Enum.filter(fn ch -> ch >= "A" and ch <= "Z" end)

264 Map.update(acc, ch, 1, &(&1 + 1))

269 # Frequency Distribution

271 Converts raw letter counts into proportions (0.0 to 1.0).
272 Normalizes the data so texts of different lengths can be compared.
273
274 ## Formula

283 %{}

289 @doc """
290 # Chi-Squared Statistic
291
292 Measures how well observed data matches expected data. A value of 0
293 means perfect agreement; larger values indicate greater divergence.
294
295 ## Formula
296
297 chi2 = sum( (observed_i - expected_i)^2 / expected_i )
298
299 ## Example
300
301 iex> CodingAdventures.Stats.Frequency.chi_squared([10, 20, 30], [20, 20, 20])
302 10.0
303 """
304 def chi_squared(observed, expected) when is_list(observed) and is_list(expected) do
305 if length(observed) != length(expected) do
306 raise ArgumentError, "Observed and expected lists must have the same length"
307 end
308
309 if observed == [] do
310 raise ArgumentError, "Lists must not be empty"
311 end
312
313 Enum.zip(observed, expected)
314 |> Enum.with_index()
315 |> Enum.reduce(0.0, fn {{obs, exp}, idx}, acc ->
316 if exp == 0 do
317 raise ArgumentError, "Expected value at index #{idx} must not be zero"
318 end
319
320 diff = obs - exp
321 acc + diff * diff / exp
322 end)
323 end

336 total = counts |> Map.values() |> Enum.sum()

338 if total == 0 do

343 letter = <<code::utf8>>

345 expected = total * Map.get(expected_freq, letter, 0.0)
